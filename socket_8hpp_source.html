385  Buffer& socket::read(size_t bytes_count, Buffer& output, std::error_code& ec) noexcept {
386  static_assert(sizeof(std::remove_pointer_t<decltype(output.data())>) == sizeof(uint8_t),
387  "socket::read can't be used with container with non-byte elements");
388 
389  output.resize(bytes_count);
390  size_t total_received = 0;
391  // Read exactly bytes_count bytes, retrying when needed.
392  while (total_received < bytes_count) {
393  size_t bytes_received =
394  implementation.read(output.data() + total_received, bytes_count - total_received, ec);
395  if (ec) break;
396  total_received += bytes_received;
397  }
398  open = (ec != error::generic::disconnected);
399 
400  return output;
401  }
